const regRef = (id: string): RegulationRef => {
  const item = MOCK_REGULATIONS.items.find(r => r.id === id);
  if (!item) throw new Error(`Missing regulation item: ${id}`);

  return {
    id: item.id,
    title: `Regulation ${item.code} - ${item.title}`,
    description: item.description ?? '',
  };
};

const roomTypes = ['Single', 'Double', 'Twin', 'Suite'];
const wings = ['A', 'B', 'C'];
const areas = ['North Wing', 'South Wing', 'East Wing', 'West Wing'];
const facilities = ['TV', 'Ensuite', 'Balcony', 'Sea View', 'Air Conditioning'];

function getRandomItem<T>(arr: T[]): T {
  return arr[Math.floor(Math.random() * arr.length)];
}
const feeEnhancementTypes = ['Sea View', 'Balcony', 'Ensuite', 'Premium Care'];
function generateRooms(count: number, clientGroups: ClientGroup[] = []): Room[] {
  const rooms: Room[] = [];
  for (let i = 1; i <= count; i++) {
    const wing = getRandomItem(wings);
    const area = getRandomItem(areas);
    const roomType = getRandomItem(roomTypes);
    const roomFacilities = facilities.filter(() => Math.random() > 0.5); // Random subset

    // Randomly decide if the room is occupied (50% chance)
    const occupied = Math.random() > 0.5;
    const occupiedClientGroup = occupied ? getRandomItem(clientGroups) : undefined;

    rooms.push({
      number: i.toString().padStart(3, '0'), // e.g., '001', '002'
      type: roomType,
      area: area,
      wing: wing,
      facilities: roomFacilities,
      occupiedClientGroup: occupiedClientGroup, // optional, undefined if not occupied
    });
  }
  return rooms;
}

function generateRoomGroups(rooms: Room[]): RoomGroup[] {
  const roomGroups: RoomGroup[] = [];

  // Group rooms by type
  const groupedByType: Record<string, Room[]> = {};
  rooms.forEach(room => {
    if (!groupedByType[room.type]) groupedByType[room.type] = [];
    groupedByType[room.type].push(room);
  });

  Object.keys(groupedByType).forEach(type => {
    const feeRules: FeeRule[] = [
      {
        careBand: 'Standard',
        dependency: 'Low',
        rate: 1000 + Math.floor(Math.random() * 500), // Base rate between 1000-1500
        enhancements: feeEnhancementTypes.map(e => ({
          type: e,
          value: Math.floor(Math.random() * 200), // Enhancement value between 0-200
        })),
      },
      {
        careBand: 'Premium',
        dependency: 'High',
        rate: 1500 + Math.floor(Math.random() * 500), // Base rate between 1500-2000
        enhancements: feeEnhancementTypes.map(e => ({
          type: e,
          value: Math.floor(Math.random() * 300), // Enhancement value between 0-300
        })),
      },
    ];

    roomGroups.push({
      name: `${type} Rooms`,
      feeRules,
    });
  });

  return roomGroups;
}
const clientEnhancementTypes = ['Extra Care', 'Special Diet', 'Mobility Assistance', 'Luxury Package'];

function generateClientGroups(): ClientGroup[] {
  return [
    {
      name: 'Standard Clients',
      enhancements: clientEnhancementTypes.slice(0, 2).map(e => ({
        type: e,
        value: Math.floor(Math.random() * 100), // 0-100
      })),
    },
    {
      name: 'Premium Clients',
      enhancements: clientEnhancementTypes.map(e => ({
        type: e,
        value: Math.floor(Math.random() * 300), // 0-300
      })),
    },
  ];
}
