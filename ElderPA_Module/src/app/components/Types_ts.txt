// Where a regulation comes from (for audit trail + future refresh).
export interface RegulationSource {
  publisher: 'CQC'| string;
  hubUrl: string;        // e.g. regulations hub page
  regulationUrl?: string; // specific regulation page
  versionDate?: string;   // optional
}

// A whole library (e.g., "Fundamental Standards")
export interface RegulationLibrary {
  id: string; // e.g. "CQC-FS"
  title: string; // "Fundamental Standards"
  source: RegulationSource;
  items: RegulationItem[]; // Reg 4..20 + appendices
}

// One regulation (or appendix) that users select in the UI.
export interface RegulationItem {
  id: string;            // e.g. "FS-REG-4", "FS-APP-1"
  code: string;          // e.g. "4", "5", "Appendix 1"
  title: string;         // e.g. "Where the service provider is an individual or partnership"
  type: 'Regulation' | 'Appendix'| 'Custom';
  description?: string;  // short summary
  appliesTo: ServiceType; // usually 'Both', but allow override
  source?: RegulationSource;

  // Key part: children that become audit questions
  subsections: RegulationClause[];
}

// A subsection/clause that becomes one audit question row.
// Supports 4(4)(a) style nesting like the CQC guidance structure. [file:29]
export interface RegulationClause {
  id: string;          // stable ID e.g. "FS-REG-4-4-a"
  label: string;       // display label e.g. "4(4)(a)"
  text: string;        // the clause text users see as the question prompt
  included?: boolean;       // selection state (UI/audit creation), default true
  defaultIncluded?: boolean; // optional default for UI (library-driven)
  guidance?: string;   // optional helper text/snippet
  appliesTo: ServiceType;

  // Optional metadata for automation / filtering
  domain?: 'Safe' | 'Effective' | 'Caring' | 'Responsive' | 'WellLed';
  tags?: string[];

  // Nested clauses if needed
  children?: RegulationClause[];
}
