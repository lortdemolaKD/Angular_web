  addChecklistQuestion() {

  loadCustomTemplate(templateId: string) {
    const template = this.customTemplates().find(t => t.id === templateId);
    if (!template) return;

    const questions: AuditQuestionInstance[] = [];

    if (template.format === 'checklist') {
      (template.data as string[]).forEach((q: string, i: number) => {
        questions.push({
          templateQuestionId: crypto.randomUUID(),
          templateType: 'custom-checklist' as any,
          regulationId: template.id,  // â† Use template ID as "regulation" ID
          regulationClauseId: `CL-${i}`,  // â† Fake clause ID
          clauseLabel: `Q${i + 1}`,  // â† "Q1", "Q2"...

          text: q,
          domain: (template.domain || 'WellLed') as any,
          score: 0,

          evidence: [],  // â† Required empty array

          evidenceSummaryText: '',
          actionRequired: 'None',
          assignedTo: '',
          targetDate: '',
          completed: 'N' as any,
          completionStatus: 'NotStarted' as any,
          defaultIncluded: true
        } as AuditQuestionInstance);
      });
    }

    else if (template.format === 'table') {
      (template.data as string[][]).forEach((row: string[], rowIdx: number) => {
        row.forEach((cell: string, colIdx: number) => {
          questions.push({
            templateQuestionId: crypto.randomUUID(),
            templateType: 'custom-table' as any,
            regulationId: template.id,
            regulationClauseId: `R${rowIdx}C${colIdx}`,
            clauseLabel: `R${rowIdx + 1}C${colIdx + 1}`,
            text: cell || `Cell ${rowIdx + 1},${colIdx + 1}`,
            domain: (template.domain || 'WellLed') as any,
            score: 0,
            evidence: [],
            // ... all other required fields with defaults
          } as AuditQuestionInstance);
        });
      });
    }


    this.questions.set([...this.questions(), ...questions]);
  }

  selectAudit(audit: AuditInstance) {
    console.log('selectAudit', audit);

    // Core audit metadata
    this.selectedAudit.set(audit);
    this.auditTitle.set((audit as any).title ?? `${audit.auditType} audit`);
    this.auditType.set(audit.auditType);

    this.departmentId.set(audit.departmentId ?? null);
    this.subDepartmentId.set(audit.subDepartmentId ?? null);

    // Questions from audit
    const auditQuestions = audit.questions ?? [];
    this.questions.set(auditQuestions);

    // Rebuild selected regs from questions (or from audit.regulations if you have that)
    const usedRegIds = new Set(
      auditQuestions
        .map(q => q.regulationId)
        .filter((id: string | null | undefined): id is string => !!id)
    );

    const allRegs = this.regulations();
    const selectedRegs = allRegs.filter(r => usedRegIds.has(r.id));
    const availableRegs = allRegs.filter(r => !usedRegIds.has(r.id));


    this.selectedRegs.set(
      [...selectedRegs].sort((a: any, b: any) =>
        String(a.code).localeCompare(String(b.code), undefined, { numeric: true })
      )
    );
    this.availableRegs.set(availableRegs);

    // Optionally, ensure each selected reg has clauses/questions (similar to dropRegulation)
    for (const reg of selectedRegs) {
      this.addRegulationClausesToAudit(reg, { includeAll: true });

      if (this.getQuestionsByRegulation(reg.id).length === 0) {
        this.questions.set([
          ...this.questions(),
          {
            templateQuestionId: crypto.randomUUID(),
            regulationId: reg.id,
            text: reg.description ?? reg.title ?? 'Regulation added',
            domain: 'WellLed',
            score: 0,
            evidence: [],
            clauseLabel: '-',
            evidenceSummaryText: '',
            actionRequired: 'None',
            assignedTo: '',
            targetDate: '',
            completed: 'N',
            defaultIncluded: true
          } as any
        ]);
      }
    }
  }

  createAuditFromTemplate(templateId: string | null) {
    if (!templateId) return;

    // Find the template from the already-loaded list:
    const template = this.templates().find(t =>
      (t as any).id === templateId || (t as any)._id === templateId
    );
    if (!template) return;

    this.selectedAudit.set(null);
    this.selectedTemplateId.set(templateId);

    this.auditTitle.set((template as any).name ?? '');
    this.auditType.set(((template as any).auditType as any) ?? 'baseline');
    this.auditDate.set(new Date().toISOString().slice(0, 10));

    const rows: AuditQuestionInstance[] = [];

    for (const section of (template as any).sections ?? []) {
      for (const q of section.questions ?? []) {
        const regId = q.regulations?.[0]?.id ?? '';
        rows.push({
          templateQuestionId: crypto.randomUUID(),
          templateType: 'regulation' as any,  // â† ADD THIS
          regulationId: this.normalizeRegulationId(regId),
          text: q.text,
          domain: q.domain ?? 'WellLed' as any,
          score: 0,
          evidence: [],
          // Add other required fields your interface needs:
          evidenceSummaryText: '',
          actionRequired: 'None',
          assignedTo: '',
          targetDate: '',
          completed: 'N' as any,
          defaultIncluded: true
        } as AuditQuestionInstance);

      }
    }

    this.questions.set(rows);

    const selectedIds = new Set(rows.map(r => r.regulationId).filter(Boolean));
    const allRegs = this.regulations();
    const selected = allRegs
      .filter(r => selectedIds.has(r.id))
      .sort((a: any, b: any) =>
        String(a.code).localeCompare(String(b.code), undefined, { numeric: true })
      );

    const available = allRegs.filter(r => !selectedIds.has(r.id));

    this.selectedRegs.set(selected);
    this.availableRegs.set(available);
  }

  private normalizeRegulationId(id: string | null | undefined): string {
    const raw = (id ?? '').trim();
    if (!raw) return raw;

    // already in library format
    if (raw.startsWith('FS-REG-')) return raw;

    // old mock audit format: Reg12 -> FS-REG-12
    const m = /^Reg(\d+)$/i.exec(raw);
    if (m) return `FS-REG-${m[1]}`;

    return raw;
  }

    const qid = this.activeEvidenceQuestionId;
    if (!audit || !qid) return;

  manageEvidence() {
    const audit = this.selectedAudit();
    const qid = this.activeEvidenceQuestionId;
    if (!audit || !qid) return;

    const q = this.questions().find(x => x.templateQuestionId === qid);
    if (!q) return;

    this.closeEvidenceMenu();

    const dialogRef = this.dialog.open(EvidenceDialog, {
      width: '760px',
      data: {
        title: 'Manage evidence',
        evidenceSummaryText: (q as any).evidenceSummaryText ?? '',
        evidence: q.evidence ?? []
      }
    });

    dialogRef.afterClosed().subscribe((result: EvidenceDialogResult | undefined) => {
      if (!result) return;

      this.auditService.patchQuestion(audit.id, qid, {
        evidence: result.evidence,
        evidenceSummaryText: result.evidenceSummaryText
      } as any).subscribe((updated) => this.selectAudit(updated));
    });
  }

  saveAudit() {
    const audit = this.selectedAudit();

    if (!audit) {
      const templateId = this.selectedTemplateId();
      if (!templateId) {
        alert('Select a template first (templateId is required).');
        return;
      }}

    // If audit exists: patch header fields only
    if (audit) {
      this.auditService.patch(audit.id, {
        auditType: this.auditType(),
        date: this.auditDate(),
        departmentId: this.departmentId() ?? undefined,
        subDepartmentId: this.subDepartmentId() ?? undefined
      }).subscribe((updated) => this.selectAudit(updated));
      return;
    }

    // If no audit yet: create empty audit (or require templateId)
    const payload: Partial<AuditInstance> = {
      title:this.auditTitle()|| undefined,
      templateId: this.selectedTemplateId() || undefined,
      auditType: this.auditType(),
      date: this.auditDate(),
      locationId: this.locationId() || undefined,
      departmentId: this.departmentId() || undefined,
      subDepartmentId: this.subDepartmentId() || undefined,
      questions: this.questions().map(q => ({
        templateQuestionId: q.templateQuestionId,
        templateType: q.templateType || 'regulation' as any,  // â† ADD
        regulationId: q.regulationId,
        text: q.text,
        domain: q.domain,
        score: q.score ?? 0,
        evidence: q.evidence ?? [],
        completionStatus: q.completionStatus ?? 'NotStarted',
        defaultIncluded: q.defaultIncluded ?? true,
        // Add other missing required fields:
        evidenceSummaryText: q.evidenceSummaryText ?? '',
        actionRequired: q.actionRequired ?? 'None',
        assignedTo: q.assignedTo ?? '',
        targetDate: q.targetDate ?? '',
        completed: q.completed ?? 'N' as any
      } as any))

    };

// Log before send
    console.log('Sending audit payload:', payload);
    this.auditService.create(payload).subscribe({
      next: created => {
        this.audits.update(audits => [created, ...audits]);
        this.selectAudit(created);
      },
      error: err => console.error('Create failed:', err.error)
    });

  private getRegulationClauses(reg: any): RegulationClause[] {
    const raw = reg.subsections ?? reg.sections ?? reg.clauses ?? [];
    return Array.isArray(raw) ? raw : [];
  }

  private flattenClauses(clauses: RegulationClause[]): RegulationClause[] {
    const out: RegulationClause[] = [];
    const walk = (c: RegulationClause) => {
      out.push(c);
      const children = c.children ?? c.subsections ?? c.items ?? [];
      if (Array.isArray(children)) children.forEach(walk);
    };
    clauses.forEach(walk);
    return out;
  }

  private findQuestionForClause(regId: string, clause: any): AuditQuestionInstance | undefined {
    const clauseId = clause.id ?? clause.clauseId ?? null;
    const clauseLabel = clause.label ?? clause.code ?? clause.clauseLabel ?? '';
    const clauseText = clause.text ?? clause.requirement ?? clause.description ?? '';

    if (clauseId) {
      return (this.questions() as any[]).find(
        q => q.regulationId === regId && (q as any).regulationClauseId === clauseId
      );
    }

    return this.questions().find(
      q => q.regulationId === regId && q.text === clauseText && (q as any).clauseLabel === clauseLabel
    );
  }

  private createQuestionFromClause(reg: RegulationItem, clause: any): AuditQuestionInstance {
    const clauseId = clause.id ?? clause.clauseId ?? crypto.randomUUID();
    const clauseLabel = clause.label ?? clause.code ?? clause.clauseLabel ?? '';
    const clauseText = clause.text ?? clause.requirement ?? clause.description ?? '(no text)';
    const domain = clause.domain ?? 'WellLed';

    return {
      templateQuestionId: crypto.randomUUID(),
      regulationId: reg.id,
      text: clauseText,
      domain,
      score: 0,
      evidence: [],
      regulationClauseId: clauseId,
      clauseLabel,
      evidenceSummaryText: '',
      actionRequired: 'None',
      assignedTo: '',
      targetDate: '',
      completed: 'N',
      defaultIncluded: clause.defaultIncluded ?? false
    } as any;
  }

  private addRegulationClausesToAudit(reg: RegulationItem, opts: { includeAll: boolean }) {
    const clauses = this.flattenClauses(this.getRegulationClauses(reg));
    if (!clauses.length) return;

    const existing = this.questions();
    const toAdd: AuditQuestionInstance[] = [];

    for (const clause of clauses) {
      const already = this.findQuestionForClause(reg.id, clause);
      if (already) continue;
      if (!opts.includeAll) continue;
      toAdd.push(this.createQuestionFromClause(reg, clause));
    }

    if (toAdd.length) this.questions.set([...existing, ...toAdd]);
  }

  async generateRecurringFromCurrentDraft() {
    const templateId = this.selectedTemplateId();
    if (!templateId) return;

    // Fetch template via AuditTemplateService.get(id)
    const template = await firstValueFrom(this.templateService.get(templateId));
    console.log('template:', template);
    const monthsAhead = this.generateMonthsAhead();            // 3 | 6 | 12
    const freq = this.recurringFrequency();                    // 'Daily' | 'Weekly' | ...
    const auditType = this.recurringTemplateAuditType();       // 'baseline' | 'registered_manager' | 'provider'

    const toIsoDate = (d: Date) => d.toISOString().slice(0, 10);

    const addPeriod = (d: Date, f: any): Date => {
      const x = new Date(d);
      switch (f) {
        case 'Daily': x.setDate(x.getDate() + 1); break;
        case 'Weekly': x.setDate(x.getDate() + 7); break;
        case 'Monthly': x.setMonth(x.getMonth() + 1); break;
        case 'Quarterly': x.setMonth(x.getMonth() + 3); break;
        case 'Annually': x.setFullYear(x.getFullYear() + 1); break;
        case 'AdHoc':
        default: x.setMonth(x.getMonth() + 1); break;
      }
      return x;
    };

    const normalizeRegulationId = (id: string | null | undefined): string => {
      const raw = (id ?? '').trim();
      if (!raw) return raw;
      if (raw.startsWith('FS-REG-')) return raw;
      const m = /^Reg(\d+)$/i.exec(raw);
      if (m) return `FS-REG-${m[1]}`;
      return raw;
    };

    const buildRowsFromTemplate = (): AuditQuestionInstance[] => {
      const rows: AuditQuestionInstance[] = [];
      for (const section of template.sections ?? []) {
        for (const q of section.questions ?? []) {
          const regId = q.regulations?.[0]?.id ?? '';
          rows.push({
            templateQuestionId: crypto.randomUUID(),
            regulationId: normalizeRegulationId(regId),
            text: q.text,
            domain: q.domain ?? 'WellLed',
            score: 0,
            evidence: [],
            evidenceSummaryText: '',
            actionRequired: 'None',
            assignedTo: '',
            targetDate: '',
            completed: 'N',
            defaultIncluded: true
          } as any);
        }
      }
      return rows;
    };

    const start = new Date(this.auditDate());
    if (Number.isNaN(start.getTime())) return;

    const until = new Date(start);
    until.setMonth(until.getMonth() + monthsAhead);

    const existing = this.audits();

    for (let d = new Date(start); d <= until; d = addPeriod(d, freq)) {
      const dateStr = toIsoDate(d);

      // Duplicate guard: date + type + location
      const already = existing.some(a =>
        a.date === dateStr &&
        a.auditType === auditType &&
        (a.locationId ?? null) === (this.locationId() ?? null)
      );
      if (already) continue;

      const payload: Partial<AuditInstance> = {
        templateId,
        title: this.auditTitle() ?? undefined,
        auditType,
        date: dateStr,
        locationId: this.locationId() ?? undefined,
        departmentId: this.departmentId() ?? undefined,
        subDepartmentId: this.subDepartmentId() ?? undefined,
        questions: buildRowsFromTemplate()
      };

      // Create via AuditService.create(...)
      await firstValueFrom(this.auditService.create(payload));
    }

    // Refresh list once
    this.loadAudits();
  }
